# Introducción a Typescript

> Tipado de datos

```ts
const a: number = 123
```

```ts
function suma(a: number, b: number) {
    return a + b
}
```

> Tipado de objetos

```ts
class Persona { ... }
type Persona = { ... }
interface Persona { ... }

const persona: Persona = ...
```

> Tipado de componentes

```ts
type ContadorProps = {
    valorInicial: number
    maximo: number
}
interface ContadorProps {
    valorInicial: number
    maximo: number
}

function Contador(props: ContadorProps) {

    // ... props.valorInicial

    return ...

}
```

> Deconstrucción de objetos

```ts
type ContadorProps = {

    valorInicial: number
    maximo?: number

}

function Contador({
    valorInicial,
    maximo
}: ContadorProps) {

    // ... valorInicial

    return ...

}
```
